if s == “”:

# 记录字符到index的映射
dic = {}
L, start_ind = 0, 0
for idx, i in enumerate(s):
    # 第idx个字符在dic中且是超过start_ind的字符
    if i in dic and start_ind <= dic[i]:
        L = max(idx - start_ind, L)
        start_ind = dic[i] + 1
    dic[i] = idx
L = max(idx + 1 - start_ind, L)
return L
